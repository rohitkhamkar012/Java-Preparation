Q3. Accept a number from user and check if it is palindrome number or not eg 
(121) 
ans-->

import java.util.Scanner;

public class Problem3 {
    static boolean checkPalin(int a)
    {
        int rev=0;
        int og=a;
        while(a!=0)
        {
            int rem=a%10;
            rev=rev*10+rem;
            a=a/10;
        }
        return rev==og;
    }
    public static void main(String[] args) {
        Scanner scanner=new Scanner(System.in);
        System.out.println("enter number");
        int a=scanner.nextInt();
        System.out.println(checkPalin(a));
    }
}



Q4. Accept a term  from user and print Fibonacci series. 
ans-->

import java.util.Scanner;

public class Problem4 {
    static void fibbo(int n)
    {
        int a,b,c;
        a=0;
        b=1;
        System.out.println(a);
        System.out.println(b);
        if(n>2)
        {
            for(int i=1;i<=n-2;i++)
            {
                c=a+b;
                System.out.println(c);
                a=b;
                b=c;
            }
        }

    }
    public static void main(String[] args) {
        Scanner scanner=new Scanner(System.in);
        System.out.println("enter number");
        int n=scanner.nextInt();
        if(n==0)
            System.out.println(0);
        // else if(n==1)
        //     System.out.println(1);
        else
            fibbo(n);
    }
}
